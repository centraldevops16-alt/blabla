#------------------ Update Tomcat registry info ------------------#
try {
  # Derive service-name based key (Tomcat10-PF-DEV_node1 or similar)
  $tomcatDirectoryParts = $directoryName.Split('\', 2)
  $regServiceName = if ($tomcatDirectoryParts.Count -gt 1) { $tomcatDirectoryParts[1] } else { $directoryName }

  $regBase = "HKLM:\SOFTWARE\Apache Software Foundation\Tomcat\10.1"

  # Key 1: service-name based (Tomcat10-PF-DEV_node1)
  $regPath1 = Join-Path -Path $regBase -ChildPath $regServiceName

  # Key 2: display-name based (Apache Tomcat 10.1 Tomcat10-PF-DEV_node1)
  $displayName = "Apache Tomcat 10.1 $serviceName"
  $regPath2 = Join-Path -Path $regBase -ChildPath $displayName

  foreach ($rp in @($regPath1, $regPath2)) {
    Write-Info "Would update registry at '$rp' with Version='$version' and InstallPath='$tomcatDir'."
    if (-not $IsDryRun) {
      if (-not (Test-Path -Path $rp)) {
        Write-Warn "Registry path '$rp' not found; creating it."
        New-Item -Path $rp -Force | Out-Null
      }
      Set-ItemProperty -Path $rp -Name 'Version'     -Value $version -Force
      Set-ItemProperty -Path $rp -Name 'InstallPath' -Value $tomcatDir -Force
    }
  }
} catch {
  Log-Error -Message ("Error updating Tomcat registry entries: {0}" -f $_.Exception.Message) -ErrorLogPath $errorLogPath
}
